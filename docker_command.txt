1) open terminal 

2) docker --version 

3) sudo apt-get update 

4) sudo apt-get install apt-transport-https ca-certificates curl software-properties-common

5) curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

6) sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable"

7) sudo apt-get install docker-ce

Build custom Docker image 
---------------------------
1) git clone https://github.com/nv284/spring_demo123.git

2) mkdir Docker 
   cd Docker 

3) docker Build -t phpcode .

4) docker images 

5) docker run -d  --name phpcode -p 80:80 phpcode

6) docker run <image name> 

7) docker run --name <container-name> <image name>


push an images to docker hub 
---------------------------------
1) mkdir docker_python
   cd docker_python/
   vi Dockerfile

Add the code given below to this Dockerfile


FROM python
WORKDIR /app
ADD . /app
RUN pip install -r requirements.txt
EXPOSE 80
ENV NAME world
CMD [“python”, “app.py”]

vi app.py

docker build -t docker_python . -f Dockerfile

docker run -d --name docker_python-p 80:80 docker_python
